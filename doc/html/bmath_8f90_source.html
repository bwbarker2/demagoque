00138 subroutine zGauss(n,a,l)
00139  !! zGauss - performs Gaussian Elimination with scaled partial pivoting as
00140  !           described in W.~Cheney, D.~Kincaid, Numerical Mathematics and
00141  !           Computing, 5th ed., Thomson (2004). Extended for complex values.
00142  !
00143  ! output: The output matrix 'a' is a triangular matrix with the rows out of
00144  ! order. The row order is specified in the array 'l'. Scaling factors used
00145  ! in the process of the Gaussian elimination are stored in 'a' in the elements
00146  ! that are eliminated. For example, if L=[3,1,2], then row 3 is the 'first'
00147  ! row, and all other rows have '0' in the first column. But instead of a '0',
00148  ! the value stored is the multiplier. In the above case, the diagonal elements
00149  ! are L(3,1), L(1,2), and L(2,3], and the row elements before the diagonal
00150  ! elements have been eliminated.
00151  !
00152  ! This format reduces the amount of redundant copying and memory needed for
00153  ! storing scaling factors needed for solving a system of linear equations.
00154  implicit none
00155 
00156  integer,                    intent(in)    :: n !dimension of square matrix
00157  complex*16, dimension(n,n), intent(inout) :: a !matrix
00158  integer,    dimension(n),   intent(out)   :: l !index array
00159 
00160  integer                  :: i,j,k  !loop variables
00161  complex*16               :: r,rmax,smax,xmult
00162  complex*16, dimension(n) :: s
00163  
00164  do i=1,n
00165    l(i)=i
00166    smax=0.d0
00167    do j=1,n
00168      if(abs(a(i,j))>abs(smax)) smax=a(i,j)
00169    enddo
00170    s(i)=smax
00171  enddo
00172 
00173  do k=1,n-1
00174    rmax=0.d0
00175    j=k
00176    do i=k,n
00177      r=a(l(i),k)/s(l(i))
00178      if(abs(r)>abs(rmax)) then
00179        rmax=r
00180        j=i
00181      endif
00182    enddo
00183    i=l(j)
00184    l(j)=l(k)
00185    l(k)=i
00186    do i=k+1,n
00187      xmult=a(l(i),k)/a(l(k),k)
00188      a(l(i),k)=xmult  !where zeroes would be, put multiplier instead
00189      do j=k+1,n
00190        a(l(i),j)=a(l(i),j)-xmult*a(l(k),j)
00191      enddo
00192    enddo
00193  enddo
00194 
00195 end subroutine zGauss
